# VM Update Commands (Docker deployment)
# Copy/paste these commands on the VM.

# 1) Pull latest code
cd ~/noc-configmaker
git pull

# 2) Rebuild + restart containers
sudo docker compose up -d --build

# 3) Verify containers are healthy
sudo docker compose ps
curl -fsS http://127.0.0.1:8000/api/health

# 4) Verify the UI locally from the VM
curl -I http://127.0.0.1:8000/app

# 5) If using system nginx for the public domain, reload it after config changes
# sudo nginx -t && sudo systemctl reload nginx
